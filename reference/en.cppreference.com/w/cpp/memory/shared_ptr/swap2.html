template< class T >
void swap( shared_ptr<T> &lhs, shared_ptr<T> &rhs );
 (since C++11)

Specializes the std::swap algorithm for std::shared_ptr. Swaps the pointers of lhs and rhs. Calls lhs.swap(rhs).
